```python
import asyncio

from typing import Dict

class RateLimiter:
    """Rate limiter –¥–ª—è —Å–æ–±–ª—é–¥–µ–Ω–∏—è –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–π Scibox"""
    
    def __init__(self, limits: Dict[str, int]):
        """
        Args:
            limits: {"qwen3-32b-awq": 2, "qwen3-coder-30b": 2, "bge-m3": 7}
        """
        self.limits = limits
        self.requests = {model: [] for model in limits.keys()}
        self.locks = {model: asyncio.Lock() for model in limits.keys()}
    
    async def acquire(self, model: str):
        """–ñ–¥—ë—Ç —Ä–∞–∑—Ä–µ—à–µ–Ω–∏—è –Ω–∞ –∑–∞–ø—Ä–æ—Å –¥–ª—è –¥–∞–Ω–Ω–æ–π –º–æ–¥–µ–ª–∏"""
        async with self.locks[model]:
            now = datetime.now()
            
            # –£–¥–∞–ª—è–µ–º –∑–∞–ø—Ä–æ—Å—ã —Å—Ç–∞—Ä—à–µ 60 —Å–µ–∫—É–Ω–¥
            self.requests[model] = [
                req_time for req_time in self.requests[model]
                if now - req_time < timedelta(seconds=60)
            ]
            
            # –ï—Å–ª–∏ –¥–æ—Å—Ç–∏–≥–ª–∏ –ª–∏–º–∏—Ç–∞, –∂–¥—ë–º
            if len(self.requests[model]) >= self.limits[model]:
                oldest = self.requests[model][0]
                wait_time = (oldest + timedelta(seconds=60) - now).total_seconds()
                
                if wait_time > 0:
                    print(f"Rate limit –¥–ª—è {model}: –∂–¥—ë–º {wait_time:.1f}s")
                    await asyncio.sleep(wait_time)
                    return await self.acquire(model)
            
            # –î–æ–±–∞–≤–ª—è–µ–º –Ω–æ–≤—ã–π –∑–∞–ø—Ä–æ—Å
            self.requests[model].append(now)
```

```python
                    if 'choices' in chunk:
                        delta = chunk['choices'][0].get('delta', {})
                        if 'content' in delta:
                            content = delta['content']
                            full_response += content
                            yield content
                except json.JSONDecodeError:
                    continue
        
        # –î–æ–±–∞–≤–ª—è–µ–º –æ—Ç–≤–µ—Ç AI –≤ –∏—Å—Ç–æ—Ä–∏—é
        self.conversation_history.append({
            "role": "assistant",
            "content": full_response
```

```python
from typing import List, Dict
import numpy as np
from .scibox_client import SciboxClient

class EmbeddingSearch:
    def __init__(self, client: SciboxClient):
        self.client = client
        self.known_solutions: List[Dict] = []  # –ë–∞–∑–∞ –∏–∑–≤–µ—Å—Ç–Ω—ã—Ö —Ä–µ—à–µ–Ω–∏–π
    
    async def add_known_solution(self, code: str, metadata: Dict):
        """–î–æ–±–∞–≤–ª—è–µ—Ç –∏–∑–≤–µ—Å—Ç–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ –≤ –±–∞–∑—É"""
        embedding = await self.client.get_embedding(code, model="bge-m3")
        
        self.known_solutions.append({
            'code': code,
            'embedding': embedding,
            'metadata': metadata
        })
    
    async def find_similar(self, code: str, threshold: float = 0.95) -> List[Dict]:
        """
        –ò—â–µ—Ç –ø–æ—Ö–æ–∂–∏–µ —Ä–µ—à–µ–Ω–∏—è —á–µ—Ä–µ–∑ cosine similarity
        
        Args:
            code: –∫–æ–¥ –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏
            threshold: –ø–æ—Ä–æ–≥ —Å—Ö–æ–∂–µ—Å—Ç–∏ (0-1)
        
        Returns:
            –°–ø–∏—Å–æ–∫ –ø–æ—Ö–æ–∂–∏—Ö —Ä–µ—à–µ–Ω–∏–π —Å similarity score
        """
        # –ü–æ–ª—É—á–∞–µ–º embedding –¥–ª—è –ø—Ä–æ–≤–µ—Ä—è–µ–º–æ–≥–æ –∫–æ–¥–∞
        code_embedding = await self.client.get_embedding(code, model="bge-m3")
        
        similar = []
        
        for solution in self.known_solutions:
            # –í—ã—á–∏—Å–ª—è–µ–º cosine similarity
            similarity = self._cosine_similarity(
                code_embedding,
                solution['embedding']
            )
            
            if similarity >= threshold:
                similar.append({
                    'code': solution['code'],
                    'similarity': similarity,
                    'metadata': solution['metadata']
                })
        
        # –°–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ —É–±—ã–≤–∞–Ω–∏—é similarity
        similar.sort(key=lambda x: x['similarity'], reverse=True)
        
        return similar
    
    @staticmethod
    def _cosine_similarity(vec1: List[float], vec2: List[float]) -> float:
        """–í—ã—á–∏—Å–ª—è–µ—Ç cosine similarity –º–µ–∂–¥—É –¥–≤—É–º—è –≤–µ–∫—Ç–æ—Ä–∞–º–∏"""
        vec1 = np.array(vec1)
        vec2 = np.array(vec2)
        
        dot_product = np.dot(vec1, vec2)
        norm_product = np.linalg.norm(vec1) * np.linalg.norm(vec2)
        
        if norm_product == 0:
            return 0.0
        
        return float(dot_product / norm_product)
```

```python
import redis
import json
from typing import Optional, Any
import hashlib

class RedisCache:
    def __init__(self, redis_url: str = "redis://localhost:6379"):
        self.redis = redis.from_url(redis_url)
    
    def _make_key(self, prefix: str, data: Any) -> str:
        """–°–æ–∑–¥–∞—ë—Ç —É–Ω–∏–∫–∞–ª—å–Ω—ã–π –∫–ª—é—á –∏–∑ –¥–∞–Ω–Ω—ã—Ö"""
        data_str = json.dumps(data, sort_keys=True)
        hash_obj = hashlib.md5(data_str.encode())
        return f"{prefix}:{hash_obj.hexdigest()}"
    
    def get_cached_task(self, level: str, domain: str) -> Optional[Dict]:
        """–ü—ã—Ç–∞–µ—Ç—Å—è –ø–æ–ª—É—á–∏—Ç—å –∑–∞–∫—ç—à–∏—Ä–æ–≤–∞–Ω–Ω—É—é –∑–∞–¥–∞—á—É"""
        key = self._make_key("task", {"level": level, "domain": domain})
        cached = self.redis.get(key)
        
        if cached:
            return json.loads(cached)
        return None
    
    def cache_task(self, level: str, domain: str, task: Dict, ttl: int = 86400):
        """–ö—ç—à–∏—Ä—É–µ—Ç –∑–∞–¥–∞—á—É –Ω–∞ 24 —á–∞—Å–∞"""
        key = self._make_key("task", {"level": level, "domain": domain})
        self.redis.setex(key, ttl, json.dumps(task))
    
    def get_cached_evaluation(self, code_hash: str) -> Optional[Dict]:
        """–ü—ã—Ç–∞–µ—Ç—Å—è –ø–æ–ª—É—á–∏—Ç—å –∑–∞–∫—ç—à–∏—Ä–æ–≤–∞–Ω–Ω—É—é –æ—Ü–µ–Ω–∫—É"""

        cached = self.redis.get(key)
        
        if cached:
            return json.loads(cached)
```
